Convertir Celsius en Fahrenheit (A)

Entrée : Température en Celsius (Number)

Sortie : Température en Fahrenheit (Number)

// La fonction convertCtoF prend un seul argument (celsius)
function convertCtoF(celsius) {
    // on déclare la variable qui aura la valeur finale en Fahrenheit
    let fahrenheit;
    // on lui assigne la valeur de l'argument multiplié par le taux de conversion
    fahrenheit = (celsius * 9/5) + 32;
    // on renvoie la valeur de la variable fahrenheit
    return fahrenheit;
}

// Appel de la fonction avec pour argument "30"
convertCtoF(30);

Renverser une phrase (B)

Entrée : Chaîne de caractère (String)

Sortie : Chaîne de caractère en sens inverse (String)

// La fonction reverseString() prend un seul argument (str)
function reverseString(str) {
    // On déclare une variable string vide
    let reversedStr ="";
    
    /* On boucle sur l'argument str, et on commence au dernier caractère (str.length - 1) pour revenir progressivement jusqu'au premier caractère. A chaque boucle on insère avec += str[i] le caractère dans la variable reversedStr, qui se construit progressivement par la fin */
     
    for(let i= str.length - 1; i >= 0; i--) {
        reversedStr += str[i];
    }
    // on renvoie la valeur de reversedStr
    return reversedStr;
    }
// Appel de la fonction reverseString()
reverseString("hello");

Trouver la factorielle d'un nombre (C)

Entrée : Nombre supérieur ou égal à 0 (Number)

Sortie : Factoriel de ce nombre (Si nombre = n alors factoriel de n! = 1 x...x n) (Number)

// Plusieurs façons de procéder : 
// Par une boucle classique
function factorialize(num) {
    //0! = 1 donc on initialise une variable product à 1
    let product = 1;
    // On boucle en incrémentant à partir de 2 jusqu'au nombre passé en argument
    // A chaque itération de la boucle product est multiplié par i (donc 2, puis 3,
    // etc... jusqu'à "num")
    for (let i = 2; i <= num; i++) {
      product *= i;
    }
    // On renvoie la valeur de product
    return product;
  }
  // On appelle la fonction avec l'argument 5
  factorialize(5);

  // Par une fonction récursive : 
  function factorialize(num) {
    //On commence par une condition : si num est égal à 0 alors 0! = 1
    if (num === 0) {
      return 1;
    }
    // Dans tous les autres cas, on fait appel à la récursivité : on renvoie le résultat de l'argument multiplié
    // par un appel à cette même fonction mais avec un argument - 1, qui elle-même fera appel à cette fonction
    // avec un argument - 1, jusqu'à ce que num = 1 et donc num - 1 = 0
    return num * factorialize(num - 1);
  }
  // On appelle la fonction avec l'argument 5
  factorialize(5);

Trouver le mot le plus long dans une phrase (D)

Entrée : Suite de mots (String)

Sortie : Mot le plus long et son nombre de caracteres(string et number)

function findLongestWord(str) {
    // On déclare une variable pour la taille du mot le plus long
    let maxLength = 0;
    //On déclare une variable pour le mot le plus long
    let biggestWord = '';
    //On déclare un tableau dans lequel mettre les deux variables précédentes
    const resultArray = [];
    // On coupe la phrase à partir des espaces et on met chaque mot dans le tableau "words"
    const words = str.split(' ');
    // On boucle sur chaque mot du tableau words
    for(let i = 0; i < words.length; i++) {
        // Quand un mot est plus long que maxLength(soit 0 au départ)
        if(words[i].length > maxLength) {
        // On change la valeur de maxLength pour celle du nouveau mot trouvé
        maxLength = words[i].length;
        // Le mot plus long est aussi enregistré dans la variable biggestWord
        biggestWord = words[i];
        }
    }
    // Après la boucle maxLength possède la taille du mot le plus long
    // Et biggestWord la valeur de ce mot en caractères (string)
    // En JavaScript, une fonction ne peut renvoyer deux valeurs, alors on les ajoute au tableau resultArray
    resultArray.push(biggestWord, maxLength);
    // Et la valeur de cette variable (sous la forme [nombre, string]) est renvoyée
    return resultArray;
    }
    // Appel de la fonction
    findLongestWord("La nuit tous les chats sont gris");

function findLargestInArrays(arr) {

    // Variable déclarée pour l'array final
    let finalArray = [];
    // Boucle pour passer à travers les différents arrays, et deuxième boucle
    // imbriquée pour passer à travers les éléments composants chaque arrays.
    for(let i = 0; i < arr.length; i++) {
        // Déclaration de la variable qui prendra successivement la valeur la plus haute
        // Déclaration de la variable qui prendra successivement la valeur la plus haute
        // Si déclarée à "0", alors dysfonctionnement avec les chiffres négatifs.
        // On lui donne la valeur du premier élément du tableau (qui sera comparé au second etc...)
        let maxNumber = arr[i][0];

        for(let j = 0; j < arr[i].length; j++) {
            // Cette condition permet à maxNumber de changer pour la valeur
            // d'un élément si cet élément est plus élevé.
            if(arr[i][j] > maxNumber) {
                maxNumber = arr[i][j];
            }
        }
        // Après avoir bouclé à l'intérieur d'un array : 
        // maxNumber possède la valeur la plus élevée de cet array
        // On la push() dans "finalArray" avant de continuer avec la boucle suivante.
        finalArray.push(maxNumber);
    }
    
    return finalArray;
    }
    // Appel de la fonction findLargestInArrays avec pour argument un array contenant 4 arrays [[], [], [], []]
    findLargestInArrays([[17, 23, 25, 12], [25, 7, 34, 48], [4, -10, 18, 21], [-72, -3, -17, -10]]);

Vérifier si un mot ou phrase finit par une suite de caractères (F)

function confirmEnding(str, target) {
    // Sur une seule ligne, on commence par faire un slice() du premier argument 'str'
    // slice() prend un seul argument qui est l'index de début pour couper
    // ici : on coupe str à partir de la longueur de str moins la longueur des caractères recherchés
    // exemple : str = 'travail' (7) et target = 'ail' (3)
    // on commence à couper à l'index 4 soit juste après 'trav'
    // de manière à vérifier que le target est bien à la fin
    // si c'est le cas, alors l'expression renvoie true, sinon False

    return str.slice(str.length -target.length) === target;
    }

Convertir un nombre en chiffres romains (K)

Entrée : Un nombre(Number)

Sortie : Chiffre romain(String)

function romanize (num) {
    // on commence par vérifier que l'argument est de type number
    // le "+" indique qu'on récupère la valeur numérique d'un nombre sous format string "234" => 234
    if (isNaN(+num)){
    // dans le cas contraire on retourne un avertissement
        return "Veuillez rentrer un nombre";
    }
    // on déclare une première variable digit qui a pour valeur l'argument num passé en string,
    // puis split en un tableau 234 => "234" => ["2", "3", "4"]
    let digit = String(+num).split("");
    // la variable key va servir à parcourir tous les caractères romains envisageables
    // sur le chiffre des unités (de key[20 - 30])
    // sur le chiffre des dizaines (de key[10-20])
    // sur le chiffre des centaines (de key[0-10])
    const key = ["","C","CC","CCC","CD","D","DC","DCC","DCCC","CM",
                "","X","XX","XXX","XL","L","LX","LXX","LXXX","XC",
                "","I","II","III","IV","V","VI","VII","VIII","IX"];
    // la variable roman est la string (vide) à laquelle
    // vont s'ajouter les caractères romains séléctionés
    // qui sera renvoyée en fin de fonction
    let roman = "";
    // i est l'indice d'incrémentation permettant de passer à travers le tableau key
    let i = 3;
    // on boucle de 3 à 0 (premier passage par la boucle i est déjà égal à 2, puis 1, puis 0)
    while (i--){
        // chaque itération permet de concaténer l'équivalent en chiffre romain au résultat précédent
        // +digit.pop() a pour valeur le chiffre des unités, puis celui des dizaines, puis celui des centaines
        // en ajoutant (i * 10) on avance dans le tableau key de 10 en 10

        // exemple : 234 
        // (premier passage par la boucle) i = 2
        // => digit = ["2", "3", "4"]
        // => digit.pop() = "4"
        // => +digit.pop() = 4
        // => +digit.pop() + (i*10) = 4 + (2*10) = 24
        // => key[+digit.pop()+ (i*10)] = key[24]
        // => key[24] = "IV"
        // => roman = key[24] + roman = key[24] + "" = "IV"

        // (second passage par la boucle) i = 1
        // => digit = ["2", "3"] (unité retirée par digit.pop())
        // => digit.pop() = "3"
        // => +digit.pop() = 3
        // => +digit.pop() + (i*10) = 3 + (1*10) = 13
        // => key[+digit.pop()+ (i*10)] = key[13]
        // => key[13] = "XXX"
        // => roman = key[13] + roman = key[13] + "IV" = "XXXIV"

        // (troisième passage par la boucle) i = 0
        // => digit = ["2"] (dizaine retirée par digit.pop())
        // => digit.pop() = "2"
        // => +digit.pop() = 2
        // => +digit.pop() + (i*10) = 2 + (0*10) = 2
        // => key[+digit.pop()+ (i*10)] = key[2]
        // => key[2] = "CC"
        // => roman = key[2] + roman = key[2] + "XXXIV" = "CCXXXIV"

        roman = (key[+digit.pop() + (i * 10)] || "") + roman;
    }
    // si le nombre en argument est inférieur à 1000
    // alors digit = [], les trois chiffres étant supprimés du tableau
    // sinon, les chiffres restants sont des milliers, dizaines de milliers, etc
    // exemple : 576 234 => digit = ["5", "7", "6", "2", "3", "4"]
    // après la boucle digit = ["5, "7", "6"]
    // Array(n) permet de créer un tableau de longueur n
    // Array(+digit.join("") + 1) va créer un tableau
    // qui dans l'exemple précédent sera de longueur 
    // digit.join("") = "576"
    // +digit.join("") = 576
    // +digit.join("") + 1 = 577
    // Array(577) = [,,,,,,,,,,,,, etc] de 577 éléments vides tous séparés par des virgules
    // avec Array(577).join("M") on forme une string à partir des éléments d'un tableau
    // avec comme séparateur entre les différents élements "M"
    // Les éléments étants vides, on obtient une string de la forme :
    // "M x 576" (car il y a toujours un élément de plus qu'il n'y a de virgule dans un tableau)
    // enfin, on renvoie cette valeur que l'on ajoute à la variable roman 
    // ce qui donne autant de "M" qu'il y a de milliers + centaines, dizaines, unités en chiffres romains.
    return Array(+digit.join("") + 1).join("M") + roman;
}

(Dé)coder un texte avec le Rot13 (L)

Entrée : un texte sans accent, chiffres ou ponctuation(string)

Sortie : le texte codé ou décodé (Rotation de 13 places)(string)

function toRot13(str) {
    // on fait passer l'argument str en majuscule
    // puis on split() chaque caractère dans un tableau
    // on map pour passer sur chaque élément
    
    const newStr = str.toUpperCase().split('').map((char) => {
        // pour chaque élément, code à la valeur UTF-16 du caractère (A-Z entre 65 et 90)
        const code =  char.charCodeAt(char);
        // ensuite 3 possibilités : 
        // soit le caractère est un caractère spécial (si code < 65 ou code > 90)
        // alors on renvoie le caractère sans le coder
        if(code > 90 || code < 65) {
        return String.fromCharCode(code);
        // soit le caractère est entre 78 et 90 (N - Z)
        // alors on retire 13 à la valeur UTF-16 et on renvoie le caractère
        // exemple N = 78 ; 78 - 13 = 65 = A
        } else if(code >= 78){
        return String.fromCharCode(code-13);
        // dans tous les autres cas (65 <= code < 78)
        // on ajoute 13 à la valeur UTF-16 et on renvoie le caractère
        // exemple B = 66; 66 + 13 = 79 = M
        } else {
        return String.fromCharCode(code+13);
        }
    }).join('');
    
    return newStr;
    }

Mettre en majuscule chaque première lettre d'une phrase (M)

Entrée : une chaîne de caractères (string)

Sortie : La même chaîne avec des majuscules au début de chaque mot(string)

function titleCase(str) {
    // on utilise plusieurs méthodes à la suite avant de renvoyer le résultat :
    return str
        // on passe str en minuscule
        .toLowerCase()
        // les ensembles de caractères séparés d'un espace sont placés dans un tableau
        .split(' ')
        // on passe à travers tous les éléments du tableau
        // pour chaque élément, on remplace
        // le premier caractère par le même caractère en majuscule
        .map((word) =>  word.replace(word.charAt(0), word.charAt(0).toUpperCase()))
        // on joint tous les éléments du tableau séparés par un espace
        .join(' ');
    }

Vérifier si une chaîne de caractère contient certaines lettres (Q)

Entrée : Deux chaînes de caractère (On vérifie à partir de la 2nd sur la 1ère)(String, string)

Sortie : Résultat de la recherche : True ou False(boolean)

function mutation(arr) {
    // pour faciliter la recherche, on passe les deux
    // éléments du tableau en minuscule dans de nouvelles variables
    const target = arr[1].toLowerCase();
    const searchArr = arr[0].toLowerCase();
    // on boucle sur la taille de la seconde chaîne de caractère
    for(let i = 0; i < target.length; i++) {
        // pour chaque caractère on vérifie qu'il existe sur la première
        // avec indexOf() si la lettre n'existe pas, -1 est renvoyé
        if(searchArr.indexOf(target[i]) === -1) {
        // auquel cas on renvoie "false"
        return false;
        }
    }
    // si aucun caractère absent (-1) n'a été trouvé on renvoie "true"
    return true;
    }

function chunkArrayInGroups(arr, size) {
    // on commence par créer un tableau pour la sortie
    const newArr = [];
    // on boucle sur la taille du tableau arr en argument
    // mais on ne veut pas incrémenter i de 1 à chaque passage
    // en incrémentant i de "size", on est sur d'avancer jusqu'au prochain groupe
    for(let i = 0; i < arr.length; i += size){
        // on ajoute à newArr la partie d'arr qui va de i jusqu'à size (1er passage)
        // puis de size jusqu'à 2xsize, 3xsize, 4xsize, etc...
        newArr.push(arr.slice(i, i +size))
    }
    return newArr;
    }

Calculer la somme de tous les nombres entre deux(inclus)

Entrée : Tableau de deux nombres(number)

Sortie : Somme (number)

function sumAll(arr) {
    // on déclare une variable pour chaque nombre du tableau
    const firstNum = arr[0];
    const lastNum = arr[1];

    // On calcule la quantité de nombres entre ces deux chiffres
    // le +1 permet d'inclure les deux nombres [a,b] sinon [a,b[ sans.
    // Math.abs() donne un résultat en valeur absolue (pas de négatifs)
    const countNum = Math.abs(firstNum - lastNum) + 1;
    
    // on utilise la formule de la somme d'une suite arithmétique
    // somme = (n*(a1 + an)) / 2
    // n étant le nombre de termes de la suite
    // a1 le premier terme, an le dernier
    const sum = ((firstNum + lastNum) * countNum) / 2;

    return sum;
    };

Trouver les éléments présents dans un seul tableau

Entrée : Deux tableaux d'éléments(deux arrays de tous types)

Sortie : Un tableau avec les éléments uniques à chaque tableau (Array)

function diffArray(arr1, arr2) {
    // on renvoie directement le résultat des opérations réalisées
    // on commence par fusionner les deux tableaux avec concat()
    // puis on filtre pour ne renvoyer que les éléments qui
    // 1) ne sont pas présents dans le tableau d'origine arr1
    // OU
    // 2) ne sont pas présents dans le tableau d'origine arr2
    return arr1
        .concat(arr2)
        .filter(item => !arr1.includes(item) || !arr2.includes(item));
}

Rechercher et détruire des valeurs dans un tableau

Entrée : Un tableau et des valeurs(Un array, et tous types de valeurs)

Sortie : Un tableau filtré (array)

// on récupère les valeurs à rechercher grâce au spread operator ...valsToRemove
// arr contient uniquement le tableau dans lequel rechercher
function destroyer(arr, ...valsToRemove) {
    // on renvoie le tableau arr filtré
    // sans les valeurs contenues dans valsToRemove
    return arr.filter(x => !valsToRemove.includes(x));
    }


Notes

Séparer les éléments du tableau par des espaces.
Séparer le tableau des valeurs à rechercher, ainsi que les valeurs entres elles, par une virgule et un espace
Exemple :
Input : "1 2 3 1 2 3, 2, 3" => [1,2,3,1,2,3], 2, 3 où 2 et 3 sont les valeurs à rechercher dans le tableau.
